-- 1Ô∏è‚É£ Drop tables if they exist
DROP TABLE IF EXISTS Library_Audit;
DROP TABLE IF EXISTS Library;

-- 2Ô∏è‚É£ Create main Library table
CREATE TABLE Library (
    book_id INT PRIMARY KEY,
    book_name VARCHAR(50),
    author VARCHAR(50),
    price DECIMAL(10,2)
);

-- 3Ô∏è‚É£ Create audit table
CREATE TABLE Library_Audit (
    audit_id INT AUTO_INCREMENT PRIMARY KEY,
    book_id INT,
    book_name VARCHAR(50),
    author VARCHAR(50),
    price DECIMAL(10,2),
    operation VARCHAR(10),
    audit_date DATETIME
);

-- 4Ô∏è‚É£ Create AFTER UPDATE trigger
DELIMITER $$

CREATE TRIGGER trg_library_update
AFTER UPDATE ON Library
FOR EACH ROW
BEGIN
    INSERT INTO Library_Audit(book_id, book_name, author, price, operation, audit_date)
    VALUES (OLD.book_id, OLD.book_name, OLD.author, OLD.price, 'UPDATE', NOW());
END$$

-- 5Ô∏è‚É£ Create AFTER DELETE trigger
CREATE TRIGGER trg_library_delete
AFTER DELETE ON Library
FOR EACH ROW
BEGIN
    INSERT INTO Library_Audit(book_id, book_name, author, price, operation, audit_date)
    VALUES (OLD.book_id, OLD.book_name, OLD.author, OLD.price, 'DELETE', NOW());
END$$

DELIMITER ;

-- 7Ô∏è‚É£ Display Library table BEFORE changes

-- 8Ô∏è‚É£ Perform Update and Delete operations

DELETE FROM Library WHERE book_id = 102;

-- 9Ô∏è‚É£ Display Library table AFTER changes
SELECT * FROM Library;
